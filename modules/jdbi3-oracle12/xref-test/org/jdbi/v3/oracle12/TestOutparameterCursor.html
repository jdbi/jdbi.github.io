14  package org.jdbi.v3.oracle12;

20  import org.jdbi.v3.core.mapper.RowMapper;

24  import org.jdbi.v3.testing.junit5.tc.JdbiTestcontainersExtension;
25  import org.junit.jupiter.api.Tag;
26  import org.junit.jupiter.api.Test;
27  import org.junit.jupiter.api.condition.EnabledOnOs;
28  import org.junit.jupiter.api.extension.RegisterExtension;
29  import org.testcontainers.containers.OracleContainer;
30  import org.testcontainers.junit.jupiter.Container;
31  import org.testcontainers.junit.jupiter.Testcontainers;
32  
33  import static org.assertj.core.api.Assertions.assertThat;
34  
35  @Tag("slow")
36  @Testcontainers
37  @EnabledOnOs(architectures = {"x86_64", "amd64"})
38  public class TestOutparameterCursor {
39  
40      static final String CONTAINER_VERSION = "gvenzl/oracle-xe:" + System.getProperty("oracle.container.version", "slim-faststart");
41  
42      @Container
43      static OracleContainer oc = new OracleContainer(CONTAINER_VERSION);
44  
45      @RegisterExtension
46      JdbiExtension oracleExtension = JdbiTestcontainersExtension.instance(oc)
47              .withInitializer((ds, h) -> {
48                  h.execute("CREATE TABLE USERS (ID INTEGER, NAME VARCHAR(255))");
49                  h.execute("INSERT INTO USERS VALUES (1, 'Alice')");
50                  h.execute("INSERT INTO USERS VALUES (2, 'Bob')");
51                  h.execute("CREATE OR REPLACE PROCEDURE get_user_by_name(\n"
52                          + "p_name IN USERS.NAME%TYPE,\n"
53                          + "o_c_dbuser OUT SYS_REFCURSOR) "
54                          + "AS\n"
55                          + "BEGIN\n"
56                          + "OPEN o_c_dbuser FOR\n"
57                          + "SELECT * FROM USERS WHERE NAME LIKE p_name || '%';\n"
58                          + "END;");
59              });
60  
61      @Test
62      public void someTest() throws Exception {
63          RowMapper<User> userMapper = ConstructorMapper.of(User.class);
64  
65          try (Call call = oracleExtension.getSharedHandle().createCall("call get_user_by_name(:a,:b)")
66                  .bind("a", "Alice")
67                  .registerOutParameter("b", Types.REF_CURSOR)) {
68              List<User> result = call.invoke(outParameters -> {
69                  return outParameters.getRowSet("b").map(userMapper).list();
70              });
71  
72              assertThat(result).isNotNull().hasSize(1).contains(new User(1, "Alice"));
73          }
74      }
75  
76      public static class User {
77  
78          private final int id;
79          private final String name;
80  
81          public User(int id, String name) {
82              this.id = id;
83              this.name = name;
84          }
85  
86          @Override
87          public boolean equals(Object o) {
88              if (this == o) {
89                  return true;
90              }
91              if (o == null || getClass() != o.getClass()) {
92                  return false;
93              }
94              User user = (User) o;
95              return id == user.id && Objects.equals(name, user.name);
96          }
97  
98          @Override
99          public int hashCode() {
100             return Objects.hash(id, name);
101         }
102     }
103 }
